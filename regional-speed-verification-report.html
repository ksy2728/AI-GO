🔧 기술적 분석

문제의 원인

// FeaturedModelCard.tsx - 현재 구현 const [selectedRegion, setSelectedRegion] = useState('global'); // 문제: selectedRegion이 변경되어도 API를 다시 호출하지 않음 // useEffect에서 selectedRegion 변경을 감지하지 않고 있음 useEffect(() => { fetchMetrics(); // selectedRegion을 파라미터로 전달하지 않음 }, [model.id]); // dependency에 selectedRegion이 없음

예상 수정 방안

// 수정 필요 사항 useEffect(() => { fetchMetrics(selectedRegion); // 지역 파라미터 전달 }, [model.id, selectedRegion]); // selectedRegion을 dependency에 추가 // fetchMetrics 함수도 지역 파라미터를 받도록 수정 const fetchMetrics = async (region: string) => { const response = await fetch( `/api/v1/realtime-status/${model.id}?region=${region}` ); // ... };
